  --font-family: system-ui, sans-serif;
  --font-family-mono: ui-monospace, 'Cascadia Code', 'Source Code Pro', Menlo, Consolas, 'DejaVu Sans Mono', monospace;
}

body {
  font-family: var(--font-family);
}

code, pre, kbd, samp {
  font-family: var(--font-family-mono);
}
